121template <NumericType pixel_type>
122Cube<pixel_type> read_img(string filename) {
123 using namespace boost::gil;
124 rgb8_image_t img;
125
126 string file_ext = get_file_extension(filename);
127
128 if (file_ext == "jpeg" || file_ext == "jpg")
129 read_and_convert_image(filename, img, jpeg_tag{});
130 else if (file_ext == "png")
131 read_and_convert_image(filename, img, png_tag{});
132 else
133 throw "input image format is not supported!";
134
135 Cube<pixel_type> storage(img.height(), img.width(), num_channels<rgb8_image_t>());
136 for_each_pixel(const_view(img), PixelReader(&storage));
137
138 return storage;
139}

69template <NumericType pixel_type>
70Cube<pixel_type> maxCrop(Cube<pixel_type> &srcImg, double img_out_ratio)
71{
72 // function to center crop img to specified aspect ratio
73
74 double srcN_h = (double)(srcImg.n_rows);
75 double srcN_w = (double)(srcImg.n_cols);
76
77 double currRatio = srcN_w / srcN_h;
78
79 // new dims before adjusting for ratio
80 // either newH or newW will be modified later
81 int newH = srcN_h;
82 int newW = srcN_w;
83 int rwStartIdx = 0;
84 int clStartIdx = 0;
85
86 if (currRatio < img_out_ratio)
87 {
88 newH = (int)(srcN_w / img_out_ratio);
89 rwStartIdx = (srcN_h - newH) / 2; // align cropped img in center
90 }
91 else
92 {
93 newW = (int)(srcN_h * img_out_ratio);
94 clStartIdx = (srcN_w - newW) / 2; // align cropped img in center
95 }
96
97 Cube<pixel_type> ratioCroppedImg(newH, newW, 3);
98 ratioCroppedImg.fill(0);
99
100 // end idx for span
101 int rwEndIdx = rwStartIdx + newH - 1;
102 int clEndIdx = clStartIdx + newW - 1;
103
104 ratioCroppedImg.subcube(0, 0, 0, newH - 1, newW - 1, 2) =
105 srcImg.subcube(rwStartIdx, clStartIdx, 0, rwEndIdx, clEndIdx, 2);
106
107 return ratioCroppedImg;

154 resizedImg.fill(0);

182template <NumericType pixel_type>
183Cube<pixel_type> getBestMatch(Cube<pixel_type> &targetImg, vector<Cube<pixel_type>> &srcImg_List){
184
185
186 int argmax = 0;
187 vector<float> targetAvgColor = getAvgColor(targetImg);
188 double smallest_ed = std::numeric_limits<double>::infinity();
189 double greatest_NCC = 0;
190 bool euclidDist = true;
191 for(int i = 0; i < (int)(srcImg_List.size()); i++){
192 vector<float> srcAvgColor = getAvgColor(srcImg_List[i]);
193
194 if(euclidDist){
195 // sum squared differnce (euclidian dist)
196 double ed = 0;
197 for (int j =0; j < (int)(targetAvgColor.size()); j++){
198 ed += pow((targetAvgColor.at(j) - srcAvgColor.at(j)),2);
199 }
200 // ed = sqrt(ed);
201 if (ed < smallest_ed){
202 smallest_ed = ed;
203 argmax = i;
204
205 }
206 } else {
207 // normalize cross correlation
208 double sumNumerator = 0;
209 double sumDenom1 = 0;
210 double sumDenom2 = 0;
211 double NCC = 0;
212
213 for (int j =0; j < (int)(targetAvgColor.size()); j++){
214 sumNumerator += (targetAvgColor.at(j)*srcAvgColor.at(j));
215 sumDenom1 += pow(targetAvgColor.at(j),2);
216 sumDenom2 += pow(srcAvgColor.at(j),2);
217 }
218 NCC = sumNumerator/sqrt(sumDenom1*sumDenom2);
219 if (NCC > greatest_NCC){
220 greatest_NCC = NCC;
221 argmax = i;
222 }
223 }
224 }
225
226
227 return srcImg_List[argmax];

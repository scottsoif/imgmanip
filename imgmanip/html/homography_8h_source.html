22template <typename rowColType>
23 mat computeHomography (Mat<rowColType>& srcPts, Mat<rowColType>& destPts){
24 int n_row = srcPts.n_rows;
25 int n_col = 9;
26 Mat<rowColType> newSrcPts(2*n_row, n_col, fill::zeros);
27 for (int i = 0; i< n_row; i++){
28 rowColType x_s = srcPts.at(i, 0);
29 rowColType y_s = srcPts.at(i, 1);
30 rowColType x_d = destPts.at(i,0);
31 rowColType y_d = destPts.at(i,1);
32 Mat<rowColType> temp = {{x_s,y_s,1,0,0,0, -x_d*x_s, -x_d*y_s, -x_d},
33 {0,0,0,x_s,y_s, 1, -y_d*x_s, -y_d*y_s, -y_d}};
34 newSrcPts.rows(2*i, 2*i+1) = temp;
35
36 }
37
38 vec D;// eigen value diagonal no need to diagnolize
39 mat V;
40 mat A = conv_to<mat>::from(newSrcPts);
41 mat finalA = (A.t() )*A;
42
43 eig_sym(D,V,finalA);
44 //double minD = min(diagvec(D));
45 int minDIdx = index_min(D);
46 mat H_3x3 = V.col(minDIdx);
47 H_3x3.reshape(3,3);
48 H_3x3 = H_3x3.t();
49
50 return H_3x3;

62template <typename rowColType>
63Mat<int> applyHomography(mat H_3x3, Mat<rowColType> srcPtsSqueezed){
64
65 Mat<rowColType> expandOnes(srcPtsSqueezed.n_rows, 1,fill::ones);
66 Mat<rowColType> srcPts = join_rows(srcPtsSqueezed, expandOnes);
67
68 mat destPtsTilda = H_3x3 * srcPts.t();
69
70 mat destPts(2, srcPtsSqueezed.n_rows, fill::zeros);
71 // doing it this way because broadcast division wasn't
72 // working well in armadillo for 2 rows
73 destPts(0, span::all) = destPtsTilda(0, span::all) / destPtsTilda(2, span::all);
74 destPts(1, span::all) = destPtsTilda(1, span::all) / destPtsTilda(2, span::all);
75
76 destPts = round(destPts);
77 Mat<int> destPts_int = conv_to<Mat<int>>::from(destPts);
